	2. 赋值
		= 
		+= -= *= /= %= 

		a += b  ==>  a = a + b

	3. 比较
	  	> >=
	  	< <=
	  	==
	  	=== 	判断两边的值是否相等 且 数据类型相等
	  	!=
	  	!== 	如果全等则为false, 其余都是true 

	4. 逻辑
		&& 		与假前, 与真后
		|| 		或假后, 或真前
		! 		真即是假, 假即是真

	5. 三元
		条件表达式 ? true环境 : false环境

	6. 流程控制
		顺序结构: 代码从上往下依次执行
		分支结构: 通过if 和 switch 来进行选择
		循环结构: 通过while, for 等循环来重复执行某些功能

		
		if (条件) 代码块

		if (条件) {
			代码块
		}

		if ( 条件表达式 ){
			true环境
		}else{
			false环境
		}

		
		if ( 条件1 ){
			true环境1
		}else if( 条件2 ){
			true环境2
		}...


		if ( 条件1 ){
			if( 条件2 ){
				if( 条件3 ){
					代码块3
				}	
			}
		}

		switch ( 标志 ){
			case '标志1': 代码块
			case '标志2': 代码块
			...
			case '标志3': 代码块
			default: 代码块
		}

		
		while ( 条件 ){
			代码块
		}


		do{
			代码块
		}while( 条件 )



		for( 初始值1; 条件2; 增量3){
				代码块4
		}
		执行顺序: 1 243 243 243 ...
				  当条件为假时, 结束for循环

		
		for( 键/索引/下标/偏移量 in 容器 ){
			代码块
		}


	7. 流程控制符
		break: 		立马终止循环/分支, 准备执行循环/分支外的代码
		continue:  立马结束当前一轮循环, 准备进入下一轮循环

	8. 函数
		方法1: 函数声明
			function 函数名( 参数, 参数, ... ){
				代码块
			}

		方法2: 表达式
			var 变量 = function( 参数, 参数, ... ){
				代码块
			}

		方法3: 函数构造
			var 变量 = new Function('参数1', '参数2', '参数3', .... '代码块')

	9. 函数引用
		调用函数  函数名()
		引用函数  变量 = 函数名
		自执行函数  (代码块)()
