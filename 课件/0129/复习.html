1. 增
		INSERT INTO 表名 VALUES() 
		INSERT INTO 表名 VALUES(),(),() 
		INSERT INTO 表名() VALUES() 
		INSERT INTO 表名() VALUES(),(),() 

	2. 删
		DELETE FROM 表名
		DELETE FROM 表名 WHERE 条件
		DELETE FROM 表名 [WHERE 条件] ORDER BY 排序依据 asc 升序默认/desc降序 LIMIT 行数

	3. 备份
		mysqldump -u 用户名 -p 数据库名 > 目标地址

	4. 导入
		mysql -u 用户名 -p 数据库名 < 来源地址

	5. 改
		UPDATE 表名 SET 字段名=值, 字段名=值, ...
		UPDATE 表名 SET 字段名=值, 字段名=值, ... WHERE 条件
