1. math
		.ceil()     进一取整
		.floor()    舍一取整
		.fabs()     绝对值
		.pow(x,y)   次方
		.sqrt() 	平方根
		.modf() 	取整数 和 小数
		.fsum() 	求和
		.copysign() 拷贝符号
		.pi 		圆周率
		.e 			自然底数


	2. time
		.time() 	时间戳1970-至今的秒数
		.mktime() 	 创建时间戳

		.localtime() 本地时间元组
		.gmtime() 	 世界时间元组

		.ctime() 	格式化时间戳	
		.asctime()  格式化时间元组

		.strftime() 将时间元组   格式化为 时间字符串
		.strptime() 将时间字符串 格式化为 时间元组

		.sleep() 	 睡眠
		.process_time() cpu运行时间, 不包含睡眠时间
		.perf_counter() 系统运行时间, 包含睡眠时间
		
	3. calendar
		.calendar() 	获取一年的日历
		.month() 		获取一个月的日历
		.isleap() 		判断是否为闰年

	4. datetime
		.now() 			 当前时间
		.fromtimestamp() 格式化时间戳
		.strftime() 	 将时间元组   格式化为 时间字符串
		.strptime() 	 将时间字符串 格式化为 时间元组
		.timedelta() 
					days 	天
					seconds 秒
					minutes 分
					hours   时
					weeks   周
					microseconds 微秒
					milliseconds 毫秒

					没有年

	5. os
		属性
		.name 		查看当前系统
		.sep 		查看当前系统的分隔符
					windows  \   /
					linux    /

		方法
		.getcwd() 	查看当前工作目录
		.mkdir() 	创建目录
		.makedirs() 创建多级目录
		.chdir() 	更换当前目录
		.rmdir() 	删除目录,  只能删除空目录
		.system()   系统命令
			windows:
					type null>文件名
					del 文件名
					mkdir 目录名
					rmdir 目录名

			linux:
					touch 文件名
					unlink 文件名
					mkdir 目录名
					rmdir 目录名
