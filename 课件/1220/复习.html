1. shutil
		.copyfile() 	复制文件内容
		.copymode() 	复制文件权限
		.copystat() 	复制文件权限 和 时间
		.copy() 		复制文件权限 和 内容
		.copy2() 		复制文件权限, 内容和时间

		.copytree() 	复制目录
		.rmtree() 		删除目录
		
		.move() 		剪切 / 重命名

	2. zipfile
		zip包 = zipfile.ZipFile() 	创建一个压缩包
		zip包.write() 
		zip包.extract()
		zip包.extractall()
		zip包.close()
